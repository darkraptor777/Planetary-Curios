					}
				}
			}
			else = {
				random_system = {
					limit = { has_star_flag = gilded_church }
					spawn_system = {

						max_distance = 40
						max_jumps = 1

				}

			else_if = {

					NOT = { has_technology = tech_vitality_boosters }

					tech = tech_vitality_boosters

				}
			}

				add_monthly_resource_mult = {
     				resource = society_research
     				value = @tier4researchreward
